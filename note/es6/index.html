# ES6笔记

# 定义变量var,let和定义常量const

# 作用域

  • var 定义变量 function作用域
    • 可以重复申明
    • 无法限制修改
    • 没有块级作用域
  • let 定义变量 块级作用域
    • 作用域内不可以重复申明
    • 申明后，作用域内可以重新赋值
  • const 定义常量 块级作用域
    • 作用域内不可以重复申明
    • 什么后的常量不能更改

# 函数和箭头函数

# 基本申明方法

普通申明

function show() {
    console.log('hello world')
}

箭头函数申明

let show1=()=>{
    console.log('hello world')
}

let show2=(a, b)=>{
    return a+b;
}
console.log(show2(5, 6));

# 简便申明

  • 如果只有一个参数，()可以省略
let show=a=>{
    return a*2;
}
  • 如果只有一个return，{}可以省略
let show=a=>a*2;
let show=(a, b)=>a+b;

# 数组排序示例

let arr = [12, 5, 9, 3, 99, 33, 88];

// 升序
arr.sort(function(n1, n2){
    return n1-n2;
})

// 升序箭头函数写法
arr.sort((n1, n2)=>n1-n2);

// 降序排序
arr.sort(function(n1, n2) {
  return n2-n1;
})

// 降序箭头函数写法
arr.sort((n1, n2)=>n2-n1);

# 函数的参数

# 参数扩展/展开

  • 收集剩余的参数：

注意：...args必须为最后一个参数

function show(a, b, ...args) {
  console.log(a, b, ...args);
}

show(2, 5, 10, 33, 666);
  • 展开数组

展开后的数组元素，不能直接赋值给一个变量

let arr1 = [1, 2, 3];
let arr2 = [5, 6, 7];

let arr = [...arr1, ...arr2];

console.log(arr);

function show(...args) {
  fn(...args);
}

function fn(a, b) {
  console.log(a+b);
}

show(5, 6);

# 默认参数

function show(a, b=5, c=6) {
  return a+b+c;
}

console.log(show(1, 2, 3));

# 解构赋值

# 解构赋值的条件

  • 左右两边的数据结构必须一致
  • 右边必须是已知的数据
  • 声明和赋值不能分开，必须在一句话完成

# 示例

  • 示例一
let [a, b, c] = [4, 5, 6];

console.log(a, b, c);
  • 示例二
let {a, b, c} = {a: 12, b: 5, c:6};
console.log(a, b, c);
  • 示例三
let [{a, b}, [c, d, e], num, str] = [{a:12, b: 6}, [1, 3, 5], 8, 'asdfsafa']
console.log(a, b, c, d, e, num, str);

# map示例

let arr = [2, 3, 5];
let res = arr.map(function (item) {
    return item*5;
})

// 简写
let res = arr.map(item=>item*5);

console.log(res)

let score = [55, 80, 90, 70];
let res = score.map(item=>item>=60?'及格':'不及格');
console.log(res);

# reduce 示例

reduce 输入多个数据出来一个结果，如算总数，求平均数

// 求数组arr的元素的和
let arr = [55, 80, 90, 70];
// tmp中间临时变量， item下一个需要处理的元素， 键
let res = arr.reduce(function (tmp, item, index) {
    return tmp+item;
})
console.log(res);

// 求数组arr的元素平均值
let arr = [55, 80, 90, 70];
let res = arr.reduce(function(tmp, item, index) {
  if (index != arr.length-1) {
      return tmp + item;
  } else {
      return (tmp + item) / arr.length;
  }
})

console.log(res);

# filter 过滤器

// 过滤能被3整除的数
let arr = [1, 3, 5, 6, 8 , 10, 20];
let res = arr.filter(function(item) {
  if (item % 3 == 0){
      return true;
  } else {
      return false;
  }
})
console.log(res);

// 能被3整除的数，简写
let arr = [1, 3, 5, 6, 8 , 10, 20];
let res = arr.filter(item=>item%3==0);
console.log(res);

// 过滤json中price>=10000的对象
let arr = [
    {'name':'女士衣服', 'price': 16888},
    {'name':'女士包包', 'price': 26888},
    {'name':'男士衣服', 'price': 68},
    {'name':'男士包包', 'price': 78}
];
let res = arr.filter(item=>item.price>=10000);
console.log(res);

# 面向对象

function User(name, age) {
  this.name = name;
  this.age = age;
}
User.prototype.showName = function() {
  return this.name;
}

User.prototype.showAge = function() {
  return this.age;
}

var obj = new User('Tom', '18');
console.log(obj.age);
console.log(obj.showName());

class User {
  constructor(name, age) {
      this.name = name;
      this.age = age;
  }
  showName(){
      return this.age;
  }
  showAge(){
      return this.name;
  }
}

let obj = new User('Jerry', 20);
console.log(obj.showAge());
console.log(obj.age);
