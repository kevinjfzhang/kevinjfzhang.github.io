# curl发送http请求

public function curl_request($url, $data=null){
    //开启curl
    $curl = curl_init();
    //设置传输地址
    curl_setopt($curl, CURLOPT_URL, $url);
    //以文件流形式返回
    curl_setopt($curl, CURLOPT_RETURNTRANSFER, 1);
    //如果$data有传值，则为post提交
    if(isset($data)){
        //以post提交
        curl_setopt($curl, CURLOPT_POST, 1);
        //设置传输数据
        curl_setopt($curl, CURLOPT_POSTFIELDS, $data);
    }
    //发送curl
    $request = curl_exec($curl);
    $tmpArr = json_encode($request,TRUE);
    if(is_array($tmpArr)){
        return $tmpArr;
    }else{
        return $request;
    }
    // 关闭资源
    curl_close($curl);
}

# tree栏目分类

<?php

class Tree {
    //无限极分类
    static public function tree($cate, $html = '├─ ', $pid = 0, $level = 0) {
        $arr = array();
        foreach ($cate as $v) {
            if ($v['parent_id'] == $pid) {
                $v['level'] = $level + 1;
                if($level<=1){
                    $v['html'] = str_repeat($html, $level);
                }else{
                    $tmp_html = '';
                    $tmp_html .= $html;
                    $tmp_html .= str_repeat('──', $level-1);
                    $v['html'] = str_replace(' ', '', $tmp_html);
                }
                $arr[] = $v;
                $arr = array_merge($arr, self::tree($cate, $html, $v['id'], $level + 1));
            }
        }
        return $arr;
    }

    static public function treeLayer($cate, $name = 'child', $pid = 0) {
        $arr = array();
        foreach ($cate as $v) {
            if ($v['parent_id'] == $pid) {
                $v[$name] = self::treeLayer($cate, $name, $v['id']);
                $arr[] = $v;
            }
        }
        return $arr;
    }

    //传递一个父级id返回所有子分类id
    static public function getChildsId($cate, $pid = 0) {
        $arr = array();
        foreach ($cate as $v) {
            if ($v['parent_id'] == $pid) {
                $arr[] = $v['id'];
                $arr = array_merge($arr, self::getChildsId($cate, $v['id']));
            }
        }
        return $arr;
    }


    //传递一个子级id返回所有父分类id
    static public function treeRoute($cate, $curCate) {
        $arr = array();
        
        foreach ($cate as $v) {
            if($v['id'] == $curCate['parent_id']){
                $arr = array_merge($arr, self::treeRoute($cate, $v));
            }
        }
        $arr[] = $curCate;
        return $arr;
    }

}
